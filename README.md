# zalgo-generator
generates zalgo text



```javascript
var Zalgo = require("zalgo-generator");

var zalgo = new Zalgo({
	amount: "mini", // 1, 2, 3 or small, medium, large, etc 
	up: true,
	middle: true,
	down: true,
})

console.log(zalgo.generate("hello"))


// h̵̟̱̟̘͚̺̪ͤͩé̹̠̭͆͟l̩̰̬̰̒̃̎͘l̟̱̙̅ǫ̜̩ͫͦͭ̑



```
